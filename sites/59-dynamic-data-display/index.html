Example Query
-- Revenue by segment, last 30 days
SELECT
  customer_segment,
  metric('revenue') AS total_revenue,
  metric('orders') AS order_count,
  round(total_revenue / order_count, 2) AS aov
FROM metrics.revenue_daily
WHERE event_date >= dateadd('day', -30, today())
GROUP BY 1
ORDER BY total_revenue DESC
-- 8ms execution  ·  0.001 credits  ·  cached
